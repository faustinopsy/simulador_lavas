Código em python 
                    import math
                    tempoAtual = 0.0
                    posicaoAtualLava = 7.0
                    posicaoAlvoLava = 10.0
                    passoTempoh = 0.0001
                    fatorAnegativo = -7.0
                    baseCexponencial = 0.9
                    logaritmoNaturalC = math.log(baseCexponencial)
                    while posicaoAtualLava < posicaoAlvoLava:
                        #                  y= 7.0×(2.0 - 0.9t)
                        #                  y= 7.0×2.0−7.0×0.9×t
                        taxaVariacaoLavadydt = fatorAnegativo * baseCexponencial ** tempoAtual * logaritmoNaturalC
                        posicaoAtualLava = posicaoAtualLava + passoTempoh * taxaVariacaoLavadydt
                        tempoAtual = tempoAtual + passoTempoh
                    print("Tempo calculado: " + str(tempoAtual))
